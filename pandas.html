def switch_str(df):
    m = df.mode()
    df = df.replace(np.nan, m[0])
    return df


def switch_numb(df):
    d = pd.to_numeric(df)
    m = d.mean()
    df = df.replace(np.nan, m)
    return df

def normalize(df):
    a = df.values
    m = a.mean()
    s = a.std()
    if s == 0:
        return df
    for i in range(len(a)):
        a[i] = (a[i] + m) / s
    df = pd.DataFrame(a)
    return a

def denormalize(df):
    a = df.values

    if s == 0:
        return df
